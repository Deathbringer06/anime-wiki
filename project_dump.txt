                Японская анимация, ставшая сегодня глобальным культурным и экономическим феноменом, прошла долгий путь от первых экспериментальных короткометражек начала XX века до масштабной индустрии с миллионной аудиторией по всему миру.
            </p>

            <h2>Первые шаги (1917–1940)</h2>
            <p>
                История аниме началась в 1917 году, когда японские художники-карикатуристы Отен Симокава, Дзюнъити Коути и Сэитаро Китаяма начали экспериментировать с техникой аппликационной и меловой анимации. Первые работы длились всего несколько минут, создавались вручную на чистом энтузиазме и часто брали за основу классические японские народные сказки.
            </p>

            <h2>Эпоха Осаму Тэдзуки (1960-е)</h2>
            <p>
                Настоящая революция в индустрии произошла благодаря Осаму Тэдзуке, которого по праву называют «богом манги и аниме». Вдохновленный работами Уолта Диснея, Тэдзука адаптировал свою мангу <i>Astro Boy</i> для телеэкранов в 1963 году. Именно он заложил те стилистические основы, которые мы узнаем и сегодня: большие выразительные глаза персонажей, использование кинематографичных ракурсов камеры и динамичный монтаж, компенсирующий ограниченный бюджет анимации.
            </p>

            <h2>Золотой век и мировое признание</h2>
            <p>
                В 1980-х годах индустрия пережила колоссальный технологический скачок. Появились такие легендарные студии, как Studio Ghibli во главе с Хаяо Миядзаки. Полнометражные фильмы студии вывели аниме на уровень высокого художественного искусства, завоевав признание критиков на международных кинофестивалях и навсегда изменив восприятие анимации как "контента исключительно для детей".
            </p>
        </article>

document.addEventListener('DOMContentLoaded', () => {
    // ==========================================================================
    // 1. ИНИЦИАЛИЗАЦИЯ И ПОИСК ЭЛЕМЕНТОВ ШАПКИ
    // ==========================================================================
    const menuToggleBtn = document.getElementById('menuToggleBtn');
    const siteNav = document.getElementById('siteNav');
    const searchToggleBtn = document.getElementById('searchToggleBtn');
    const searchDropdown = document.getElementById('searchDropdown');
    const toast = document.getElementById('toast');

    // ==========================================================================
    // 2. ЛОГИКА ВЫПАДАЮЩИХ ПАНЕЛЕЙ (МЕНЮ И ПОИСК)
    // ==========================================================================
    
    // Переключение мобильного меню навигации
    if (menuToggleBtn && siteNav) {
        menuToggleBtn.addEventListener('click', (e) => {
            e.stopPropagation();
            siteNav.classList.toggle('hidden');
            
            // Если открываем меню, скрываем панель поиска, чтобы не мешала
            if (!siteNav.classList.contains('hidden') && searchDropdown) {
                searchDropdown.classList.add('hidden');
            }
        });
    }

    // Переключение выпадающей панели быстрого поиска
    if (searchToggleBtn && searchDropdown) {
        searchToggleBtn.addEventListener('click', (e) => {
            e.stopPropagation();
            searchDropdown.classList.toggle('hidden');
            
            // Если открываем поиск, скрываем меню навигации
            if (!searchDropdown.classList.contains('hidden') && siteNav) {
                siteNav.classList.add('hidden');
                // Автоматически ставим фокус в инпут для удобства ввода
                const input = searchDropdown.querySelector('.search-input');
                if (input) input.focus();
            }
        });
    }

    // Клик за пределами шапки автоматически закрывает меню и поиск
    document.addEventListener('click', (e) => {
        if (siteNav && !siteNav.classList.contains('hidden') && !siteNav.contains(e.target) && e.target !== menuToggleBtn) {
            siteNav.classList.add('hidden');
        }
        if (searchDropdown && !searchDropdown.classList.contains('hidden') && !searchDropdown.contains(e.target) && e.target !== searchToggleBtn) {
            searchDropdown.classList.add('hidden');
        }
    });

    // ==========================================================================
    // 3. АВТОМАТИЧЕСКАЯ ПОДСВЕТКА АКТИВНОЙ ССЫЛКИ
    // ==========================================================================
    const currentUrl = window.location.pathname;
    const navLinks = document.querySelectorAll('.site-nav a, .chip-row a');

    navLinks.forEach(link => {
        // Убираем старый класс активности со всех ссылок
        link.classList.remove('active');
        
        const hrefAttr = link.getAttribute('href');
        if (!hrefAttr) return;

        // Нормализуем пути для точного сравнения (убираем относительные переходы вроде ../)
        const cleanHref = hrefAttr.replace(/\.\.\//g, '');
        
        // Проверяем, содержит ли текущий URL имя файла из ссылки
        if (currentUrl.includes(cleanHref) && cleanHref !== '') {
            link.classList.add('active');
        } else if (currentUrl.endsWith('/') && (cleanHref === 'index.html' || cleanHref === '')) {
            // Если открыт корень сайта, подсвечиваем "Главную"
            if (link.textContent.trim() === 'Главная') {
                link.classList.add('active');
            }
        }
    });

    // ==========================================================================
    // 4. КОМПОНЕНТ УВЕДОМЛЕНИЙ (TOAST)
    // ==========================================================================
    window.showToast = function(message, duration = 3000) {
        if (!toast) return;
        
        toast.textContent = message;
        toast.classList.add('show');
        
        setTimeout(() => {
            toast.classList.remove('show');
        }, duration);
    };

    // Автоматическая оптимизация картинок на лету (из твоего оригинального кода)
    const images = document.querySelectorAll('img');
    images.forEach(img => {
        img.loading = 'lazy'; // Ленивая загрузка для экономии мобильного трафика
        if (!img.alt) {
            const fileName = img.src.split('/').pop().split('.')[0];
            img.alt = decodeURIComponent(fileName) || 'Изображение вики';
        }
    });

// СМЕНА ТЕМЫ ОФОРМЛЕНИЯ (DARK/LIGHT MODE)

function initTheme() {
    // Проверяем, сохранил ли пользователь тему ранее
    const savedTheme = localStorage.getItem('theme');
    
    // Если сохранена светлая тема, накатываем класс на body
    if (savedTheme === 'light') {
        document.body.classList.add('light-theme');
        updateThemeButtonIcon('light');
    } else {
        updateThemeButtonIcon('dark');
    }
}

window.toggleTheme = function() {
    const isLight = document.body.classList.toggle('light-theme');
    
    if (isLight) {
        localStorage.setItem('theme', 'light');
        updateThemeButtonIcon('light');
        window.showToast('Включена светлая тема ☀️');
    } else {
        localStorage.setItem('theme', 'dark');
        updateThemeButtonIcon('dark');
        window.showToast('Включена темная тема 🌙');
    }
};

function updateThemeButtonIcon(theme) {
    const themeBtn = document.getElementById('themeToggleBtn');
    if (!themeBtn) return;
    
    // Меняем иконку на кнопке в зависимости от активной темы
    themeBtn.innerHTML = theme === 'light' ? '🌙' : '☀️';
    themeBtn.setAttribute('aria-label', theme === 'light' ? 'Включить темную тему' : 'Включить светлую тему');
}

// Запускаем инициализацию темы при каждой загрузке скрипта
document.addEventListener('DOMContentLoaded', initTheme);

// Структура базы данных в localStorage:
// 'anime_tracker' -> { 'Тайтл': { status: 'watching', episodes: 4, total: 24 } }

// 1. Изменение статуса или добавление аниме в список трекинга
window.updateAnimeTracker = function(animeTitle, newStatus, totalEps = 12) {
    let tracker = JSON.parse(localStorage.getItem('anime_tracker')) || {};
    
    if (!tracker[animeTitle]) {
        // Если тайтла еще нет в трекере, инициализируем его
        tracker[animeTitle] = {
            status: newStatus,
            episodes: 0,
            total: totalEps
        };
    } else {
        // Если есть, просто обновляем статус
        tracker[animeTitle].status = newStatus;
        // Если перевели в "Просмотрено", автоматически заполняем все серии
        if (newStatus === 'completed') {
            tracker[animeTitle].episodes = tracker[animeTitle].total;
        }
    }
    
    localStorage.setItem('anime_tracker', JSON.stringify(tracker));
    window.showToast(`Статус «${animeTitle}» обновлен! 📝`);
    
    // Обновляем счетчики на карточках, если мы в каталоге
    if (typeof window.syncCatalogProgress === 'function') window.syncCatalogProgress();
};

// 2. Индикатор прогресса в один клик (+1 серия)
window.incrementEpisode = function(animeTitle, event) {
    if (event) event.stopPropagation(); // Стопаем переход по ссылке карточки
    
    let tracker = JSON.parse(localStorage.getItem('anime_tracker')) || {};
    
    // Если аниме еще не отслеживается, принудительно добавляем его в список "Смотрю"
    if (!tracker[animeTitle]) {
        tracker[animeTitle] = { status: 'watching', episodes: 1, total: 12 };
    } else {
        if (tracker[animeTitle].episodes < tracker[animeTitle].total) {
            tracker[animeTitle].episodes++;
            // Если досмотрел до конца — автоматом переводим в статус "Просмотрено"
            if (tracker[animeTitle].episodes === tracker[animeTitle].total) {
                tracker[animeTitle].status = 'completed';
                window.showToast(`Поздравляем! Вы посмотрели «${animeTitle}» 🎉`);
            }
        } else {
            window.showToast(`Уже просмотрены все серии!`);
            return;
        }
    }
    
    localStorage.setItem('anime_tracker', JSON.stringify(tracker));
    if (typeof window.syncCatalogProgress === 'function') window.syncCatalogProgress();
    
    // Если мы на странице профиля, перерисовываем текущую вкладку
    const activeTab = document.querySelector('.tab-btn.active');
    if (activeTab) {
        const currentCategory = activeTab.getAttribute('data-category');
        if (['watching', 'completed', 'plan_to_watch', 'on_hold', 'dropped'].includes(currentCategory)) {
            window.renderTrackerTab(currentCategory);
        }
    }
};

// 3. Синхронизация отображения прогресса прямо внутри каталога
window.syncCatalogProgress = function() {
    let tracker = JSON.parse(localStorage.getItem('anime_tracker')) || {};
    const progressBadges = document.querySelectorAll('.catalog-progress-inline');
    
    progressBadges.forEach(badge => {
        const title = badge.getAttribute('data-anime-title');
        if (tracker[title]) {
            const current = tracker[title].episodes;
            const total = tracker[title].total;
            badge.innerHTML = `🍿 Эпизоды: <b>${current}/${total}</b> <button class="plus-one-btn" onclick="window.incrementEpisode('${title}', event)">+1</button>`;
        } else {
            badge.innerHTML = `<button class="start-track-btn" onclick="window.incrementEpisode('${title}', event)">➕ Начать просмотр</button>`;
        }
    });
};

// 4. Логика переключения 5 вкладок на странице профиля
window.switchProfileTabNew = function(element, statusCategory) {
    const tabs = document.querySelectorAll('.tab-btn');
    tabs.forEach(tab => tab.classList.remove('active'));
    element.classList.add('active');
    
    window.renderTrackerTab(statusCategory);
};

// 5. Отрисовка списков в профиле с учетом статистики
window.renderTrackerTab = function(statusCategory) {
    const container = document.getElementById('savedListContainer');
    if (!container) return;
    
    let tracker = JSON.parse(localStorage.getItem('anime_tracker')) || {};
    container.innerHTML = '';
    
    let count = 0;
    
    for (let title in tracker) {
        if (tracker[title].status === statusCategory) {
            count++;
            const itemHTML = `
                <div class="saved-item">
                    <div style="display: flex; flex-direction: column; gap: 4px;">
                        <span class="saved-item-title">${title}</span>
                        <span style="font-size: 13px; color: var(--text-muted)">Серии: ${tracker[title].episodes} / ${tracker[title].total}</span>
                    </div>
                    <div style="display: flex; gap: 8px; align-items: center;">
                        <button class="plus-one-btn" onclick="window.incrementEpisode('${title}', event)">+1</button>
                        <button class="btn-delete" onclick="window.deleteTrackerItem('${title}', '${statusCategory}')" aria-label="Удалить">🗑️</button>
                    </div>
                </div>
            `;
            container.insertAdjacentHTML('beforeend', itemHTML);
        }
    }
    
    if (count === 0) {
        container.innerHTML = `<div class="empty-state"><p>В этой категории пока ничего нет 💨</p></div>`;
    }
    
    // Обновляем общие цифры статистики на странице профиля
    window.updateProfileStats();
};

// 6. Удаление из трекера
window.deleteTrackerItem = function(title, currentCategory) {
    let tracker = JSON.parse(localStorage.getItem('anime_tracker')) || {};
    delete tracker[title];
    localStorage.setItem('anime_tracker', JSON.stringify(tracker));
    window.showToast('Удалено из списков 🗑️');
    window.renderTrackerTab(currentCategory);
};

// 7. Подсчет общей статистики для профиля
window.updateProfileStats = function() {
    let tracker = JSON.parse(localStorage.getItem('anime_tracker')) || {};
    let stats = { watching: 0, completed: 0, plan_to_watch: 0, dropped: 0 };
    
    for (let title in tracker) {
        if (stats[tracker[title].status] !== undefined) {
            stats[tracker[title].status]++;
        }
    }
    
    // Подставляем цифры в HTML, если элементы существуют
    if(document.getElementById('stat-watching')) document.getElementById('stat-watching').innerText = stats.watching;
    if(document.getElementById('stat-completed')) document.getElementById('stat-completed').innerText = stats.completed;
    if(document.getElementById('stat-plan')) document.getElementById('stat-plan').innerText = stats.plan_to_watch;
    if(document.getElementById('stat-dropped')) document.getElementById('stat-dropped').innerText = stats.dropped;
};

// Автоматически запускаем синхронизацию прогресса в каталогах при загрузка страниц
document.addEventListener('DOMContentLoaded', () => {
    if (typeof window.syncCatalogProgress === 'function') window.syncCatalogProgress();
    if (typeof window.updateProfileStats === 'function') window.updateProfileStats();
});
